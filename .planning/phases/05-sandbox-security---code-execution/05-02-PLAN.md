- Read the user's data file from disk:
  ```python

  data_file = None
  data_filename = None
  if file_path:
      import os
      data_filename = os.path.basename(file_path)
      with open(file_path, "rb") as f:
          data_file = f.read()
  ```

- Create E2BSandboxRuntime and execute via asyncio.to_thread() (E2B SDK is synchronous, MUST not block event loop -- see RESEARCH.md Pitfall 2):
  ```python
  runtime = E2BSandboxRuntime(timeout_seconds=settings.sandbox_timeout_seconds)

  result: ExecutionResult = await asyncio.to_thread(
      runtime.execute,
      code=state.get("generated_code", ""),
      timeout=float(settings.sandbox_timeout_seconds),
      data_file=data_file,
      data_filename=data_filename,
  )
  ```

- Handle result:
  - If `result.success` (no error):
    - Build execution_result string from stdout: `"\n".join(result.stdout)` if stdout exists, else check results list for display data.
    - Return `{"execution_result": execution_result_str}`
  - If `result.error` (execution failed):
    - Check error_count against max retries (`settings.sandbox_max_retries`):
      - If retries remaining: Route to coding_agent with error context using `Command`:
        ```python
        error_msg = f"{result.error['name']}: {result.error['value']}"
        new_error_count = state.get("error_count", 0) + 1
        writer({
            "type": "retry",
            "event": "retry",
            "message": f"Execution failed: {error_msg}. Retrying with adjusted code...",
            "attempt": new_error_count,
            "max_attempts": state.get("max_steps", 3)
        })
        return Command(
            goto="coding_agent",
            update={
                "execution_result": f"Execution error: {error_msg}",
                "validation_errors": [f"Execution error: {error_msg}. Please fix the code."],
                "error_count": new_error_count,
            },
        )
        ```
      - If retries exhausted: Route to halt node:
        ```python
        return Command(
            goto="halt",
            update={
                "execution_result": f"Execution error: {error_msg}",
                "error_count": new_error_count,
                "error": "execution_failed",
            },
        )
        ```

- IMPORTANT: The execute_in_sandbox function return type must be `dict | Command[Literal["coding_agent", "halt"]]` because it can either return a state update dict (success) OR a Command for routing (failure with retry/halt). Import `Literal` from `typing` if not already imported.

- Update the `execute_in_sandbox` function signature:
  ```python

      state: ChatAgentState,
  ) -> dict | Command[Literal["coding_agent", "halt"]]:
  ```

The existing halt_node already returns a user-friendly message. Update it to match the CONTEXT.md decision for exhausted retries:
- When `error == "execution_failed"`, use the message: "Unable to complete analysis. Try: filtering data first, breaking into simpler questions, or rephrasing query."
- When `error == "max_retries_exceeded"` (validation failures), keep existing message.

Modify halt_node:
```python
async def halt_node(state: ChatAgentState) -> dict:
    writer = get_stream_writer()
    error_type = state.get("error", "")
    max_steps = state.get("max_steps", 3)

    if error_type == "execution_failed":
        writer({
            "type": "error",
            "event": "error",
            "message": "Unable to complete analysis after multiple attempts"
        })
        final_response = (
            "Unable to complete analysis. "
            "Try: filtering data first, breaking into simpler questions, or rephrasing your query."
        )
    else:
        writer({
            "type": "error",
            "event": "error",
            "message": f"Unable to generate valid code after {max_steps} attempts"
        })
        validation_errors = state.get("validation_errors", [])
        error_summary = ", ".join(validation_errors) if validation_errors else "Unknown issues"
        final_response = (
            f"I was unable to generate valid code after {max_steps} attempts. "
            f"Errors encountered: {error_summary}. "
            f"Please try rephrasing your question or providing more context."
        )

    return {
        "final_response": final_response,
        "error": error_type or "max_retries_exceeded",
    }
```
