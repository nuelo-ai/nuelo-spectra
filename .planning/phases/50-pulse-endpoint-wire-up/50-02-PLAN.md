**Endpoint 1 — POST trigger:**
```python
@router.post(
    "/{collection_id}/pulse",
    response_model=PulseRunTriggerResponse,
    status_code=status.HTTP_202_ACCEPTED,
)
async def trigger_pulse(
    collection_id: UUID,
    body: PulseRunCreate,
    current_user: WorkspaceUser,
    db: DbSession,
) -> PulseRunTriggerResponse:
    """Trigger Pulse detection on selected files in a collection.

    Returns 202 Accepted immediately; frontend polls GET pulse-runs/{run_id} for status.
    Returns 402 if insufficient credits (with required/available balance).
    Returns 409 if a detection run is already active (with active_run_id for resumption).
    """
    # Ownership verification
    collection = await CollectionService.get_user_collection(db, collection_id, current_user.id)
    if collection is None:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Collection not found")

    # Delegate to PulseService (handles 402/409 raises, credit deduction, background task)
    pulse_run = await PulseService.run_detection(
        db,
        collection_id=collection_id,
        user_id=current_user.id,
        file_ids=body.file_ids,
        user_context=body.user_context,
    )

    return PulseRunTriggerResponse(
        pulse_run_id=pulse_run.id,
        status=pulse_run.status,
        credit_cost=pulse_run.credit_cost,
    )
```

**Endpoint 2 — GET polling:**
```python
@router.get(
    "/{collection_id}/pulse-runs/{run_id}",
    response_model=PulseRunDetailResponse,
)
async def get_pulse_run(
    collection_id: UUID,
    run_id: UUID,
    current_user: WorkspaceUser,
    db: DbSession,
) -> PulseRunDetailResponse:
    """Poll a Pulse detection run for status and results.

    Returns full Signal objects in the signals list when status='completed'.
    Returns error_message when status='failed'.
    Verifies collection ownership to prevent cross-user PulseRun access.
    """
    # Ownership verification on the collection
    collection = await CollectionService.get_user_collection(db, collection_id, current_user.id)
    if collection is None:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Collection not found")

    pulse_run = await PulseService.get_pulse_run(db, run_id)
    if pulse_run is None:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Pulse run not found")

    # Cross-check: ensure run belongs to this collection (prevents ownership bypass)
    if pulse_run.collection_id != collection_id:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Pulse run not found")

    return PulseRunDetailResponse.model_validate(pulse_run)
```

```python
async def process_orphan_refunds():
    """Periodic job: find stuck PulseRuns and refund credits.

    A PulseRun is "orphaned" when:
    - status is still in ('pending', 'profiling', 'analyzing')
    - created_at is older than PULSE_ORPHAN_TIMEOUT_MINUTES
    - completed_at is NULL
    - no Signal rows exist for this run

    For each orphaned run: refund credits to the collection owner, set
    status='failed', set error_message to a timeout message.

    Uses WITH FOR UPDATE SKIP LOCKED to prevent double-refund if two
    scheduler instances fire concurrently (or if _run_pipeline exception
    handler also attempts refund simultaneously).
    """
    from app.config import get_settings
    settings = get_settings()
    timeout_minutes = settings.pulse_orphan_timeout_minutes
    cutoff = datetime.now(timezone.utc) - timedelta(minutes=timeout_minutes)

    logger.info("Starting orphan refund check (cutoff: %s min)", timeout_minutes)
    refund_count = 0

    async with async_session_maker() as db:
        try:
            # Join Collection to get user_id without N+1
            result = await db.execute(
                select(PulseRun, Collection.user_id)
                .join(Collection, PulseRun.collection_id == Collection.id)
                .where(
                    PulseRun.status.in_(["pending", "profiling", "analyzing"]),
                    PulseRun.created_at < cutoff,
                    PulseRun.completed_at.is_(None),
                )
                .with_for_update(skip_locked=True)
            )
            rows = result.all()

            for pulse_run, user_id in rows:
                # Verify no signals exist (avoid refunding partially-completed runs)
                signal_count_result = await db.execute(
                    select(func.count(Signal.id)).where(
                        Signal.pulse_run_id == pulse_run.id
                    )
                )
                signal_count = signal_count_result.scalar_one()

                if signal_count == 0:
                    await CreditService.refund(
                        db,
                        user_id,
                        pulse_run.credit_cost,
                        reason="Pulse orphan timeout: no signals produced within timeout window",
                    )
                    refund_count += 1

                pulse_run.status = "failed"
                pulse_run.error_message = (
                    f"Detection timed out after {timeout_minutes} minutes with no results"
                )

            await db.commit()
            logger.info("Orphan refund check complete. Refunded %d run(s).", refund_count)

        except Exception:
            logger.exception("Error during orphan refund processing")
            await db.rollback()
```
