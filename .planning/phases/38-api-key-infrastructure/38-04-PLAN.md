```typescript
import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import apiClient from "@/lib/apiClient"; // confirm import path from useSettings.ts

export interface ApiKeyListItem {
  id: string;
  name: string;
  description: string | null;
  key_prefix: string;
  is_active: boolean;
  created_at: string;
  last_used_at: string | null;
}

export interface ApiKeyCreateResponse {
  id: string;
  name: string;
  key_prefix: string;
  full_key: string;  // Only present on create response
  created_at: string;
}

export interface ApiKeyCreateRequest {
  name: string;
  description?: string;
}

export function useApiKeys() {
  return useQuery<ApiKeyListItem[]>({
    queryKey: ["api-keys"],
    queryFn: async () => {
      const response = await apiClient.get("/api/v1/keys");
      return response.data;
    },
  });
}

export function useCreateApiKey() {
  const queryClient = useQueryClient();
  return useMutation<ApiKeyCreateResponse, Error, ApiKeyCreateRequest>({
    mutationFn: async (data) => {
      const response = await apiClient.post("/api/v1/keys", data);
      return response.data;
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["api-keys"] });
    },
  });
}

export function useRevokeApiKey() {
  const queryClient = useQueryClient();
  return useMutation<void, Error, string>({  // string = key id
    mutationFn: async (keyId) => {
      await apiClient.delete(`/api/v1/keys/${keyId}`);
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["api-keys"] });
    },
  });
}
```
