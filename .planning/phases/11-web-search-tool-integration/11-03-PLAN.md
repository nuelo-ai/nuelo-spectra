**2. Create `frontend/src/hooks/useSearchToggle.ts`**:
```typescript
import { useState, useEffect, useCallback } from "react";
import { apiClient } from "@/lib/api-client";

interface SearchConfig {
  available: boolean;
  quota_remaining: number;
  quota_limit: number;
  quota_exceeded: boolean;
}

export function useSearchToggle() {
  const [enabled, setEnabled] = useState(false);
  const [config, setConfig] = useState<SearchConfig | null>(null);
  const [loading, setLoading] = useState(true);
  const [quotaExhausted, setQuotaExhausted] = useState(false);

  // Fetch search config on mount
  useEffect(() => {
    async function fetchConfig() {
      try {
        const res = await apiClient.get("/search/config");
        if (res.ok) {
          const data: SearchConfig = await res.json();
          setConfig(data);
          if (data.quota_exceeded) {
            setQuotaExhausted(true);
          }
        }
      } catch {
        // Search config unavailable — treat as not configured
        setConfig({ available: false, quota_remaining: 0, quota_limit: 0, quota_exceeded: true });
      } finally {
        setLoading(false);
      }
    }
    fetchConfig();
  }, []);

  const toggle = useCallback((value: boolean) => {
    if (!quotaExhausted && config?.available) {
      setEnabled(value);
    }
  }, [quotaExhausted, config?.available]);

  // Reset toggle (called after send)
  const reset = useCallback(() => {
    setEnabled(false);
  }, []);

  // Mark quota as exhausted (called when search_quota_exceeded event received)
  const markQuotaExhausted = useCallback(() => {
    setQuotaExhausted(true);
    setEnabled(false);
  }, []);

  // Refresh config (e.g., after a query completes to update remaining quota)
  const refreshConfig = useCallback(async () => {
    try {
      const res = await apiClient.get("/search/config");
      if (res.ok) {
        const data: SearchConfig = await res.json();
        setConfig(data);
        if (data.quota_exceeded) {
          setQuotaExhausted(true);
          setEnabled(false);
        }
      }
    } catch {
      // Ignore refresh failures
    }
  }, []);

  return {
    enabled,
    toggle,
    reset,
    available: config?.available ?? false,
    quotaRemaining: config?.quota_remaining ?? 0,
    quotaLimit: config?.quota_limit ?? 0,
    quotaExhausted,
    markQuotaExhausted,
    refreshConfig,
    loading,
  };
}
```
