import { apiClient } from "@/lib/api-client";

export interface CreditCosts {
  chat: number;

export function useCreditCosts() {
  return useQuery<CreditCosts>({
    queryKey: ["credit-costs"],
    queryFn: async () => {
      const res = await apiClient.get("/credit-costs");

}
