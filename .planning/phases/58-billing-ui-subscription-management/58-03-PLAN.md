**1. Create `frontend/src/components/settings/SettingsNav.tsx`:**

A tab-style navigation component using shadcn Tabs visual styling but with Next.js Link navigation (not Radix controlled tabs, since each tab is a separate route).

```typescript
"use client";

import Link from "next/link";
import { usePathname } from "next/navigation";
import { cn } from "@/lib/utils";

const TABS = [
  { label: "Profile", href: "/settings" },
  { label: "API Keys", href: "/settings/keys" },
  { label: "Plan & Billing", href: "/settings/plan", matchAlso: ["/settings/billing"] },
];

export function SettingsNav() {
  const pathname = usePathname();

  function isActive(tab: typeof TABS[number]) {
    if (pathname === tab.href) return true;
    // Use exact match or prefix + "/" to avoid false matches on hypothetical
    // routes like /settings/billing-something
    if (tab.matchAlso?.some((p) => pathname === p || pathname.startsWith(p + "/"))) return true;
    return false;
  }

  return (
    <div className="border-b">
      <nav className="flex gap-0 -mb-px overflow-x-auto">
        {TABS.map((tab) => (
          <Link
            key={tab.href}
            href={tab.href}
            className={cn(
              "px-4 py-2 text-sm font-medium border-b-2 transition-colors whitespace-nowrap",
              isActive(tab)
                ? "border-primary text-foreground"
                : "border-transparent text-muted-foreground hover:text-foreground hover:border-muted-foreground/30"
            )}
          >
            {tab.label}
          </Link>
        ))}
      </nav>
    </div>
  );
}
```

**1. Create `frontend/src/hooks/usePlanPricing.ts`:**

```typescript
"use client";

import { useQuery } from "@tanstack/react-query";
import { apiClient } from "@/lib/api-client";

export interface PlanInfo {
  tier_key: string;
  display_name: string;
  price_cents: number;
  price_display: string;
  credit_allocation: number;
  features: string[];
  is_popular: boolean;
}

export interface PlanPricingData {
  plans: PlanInfo[];
  current_tier: string;
}

export function usePlanPricing() {
  return useQuery<PlanPricingData>({
    queryKey: ["subscriptions", "plans"],
    queryFn: async () => {
      const res = await apiClient.get("/subscriptions/plans");
      if (!res.ok) throw new Error("Failed to fetch plan pricing");
      return res.json();
    },
    staleTime: 60000,
  });
}
```

**3. Create `frontend/src/components/billing/PlanCard.tsx`:**

Per UI-SPEC: Card with tier name (20px semibold), price (28px semibold), features list, action button. Basic card gets `border-primary` (3px left border) and "Most Popular" badge.

CRITICAL: Per CONTEXT.md locked decision, upgrade/downgrade (hasSubscription branch) MUST show a confirmation dialog (AlertDialog) before calling /subscriptions/change. The dialog explains timing and charge:
- Upgrade: "You'll be charged the difference immediately. New credits added now."
- Downgrade: "Change takes effect at end of billing cycle on [date]."

```typescript
"use client";

import { useState } from "react";
import { useRouter } from "next/navigation";
import { useQueryClient } from "@tanstack/react-query";
import { toast } from "sonner";
import { Card, CardHeader, CardContent } from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";
import { Button } from "@/components/ui/button";
import {
  AlertDialog, AlertDialogAction, AlertDialogCancel, AlertDialogContent,
  AlertDialogDescription, AlertDialogFooter, AlertDialogHeader, AlertDialogTitle,
  AlertDialogTrigger,
} from "@/components/ui/alert-dialog";
import { Loader2, Check } from "lucide-react";
import { apiClient } from "@/lib/api-client";
import type { PlanInfo } from "@/hooks/usePlanPricing";
import type { SubscriptionStatus } from "@/hooks/useSubscription";

interface PlanCardProps {
  plan: PlanInfo;
  isCurrentPlan: boolean;
  hasSubscription: boolean;
  currentTierKey: string;
  subscription: SubscriptionStatus | undefined;
}

export function PlanCard({ plan, isCurrentPlan, hasSubscription, currentTierKey, subscription }: PlanCardProps) {
  const [isLoading, setIsLoading] = useState(false);
  const router = useRouter();
  const queryClient = useQueryClient();

  // Determine upgrade vs downgrade for subscribed users
  const tierOrder: Record<string, number> = { on_demand: 0, standard: 1, premium: 2 };
  const currentRank = tierOrder[currentTierKey] ?? 0;
  const newRank = tierOrder[plan.tier_key] ?? 0;
  const isUpgrade = newRank > currentRank;

  const periodEnd = subscription?.current_period_end
    ? new Date(subscription.current_period_end).toLocaleDateString("en-US", { month: "short", day: "numeric", year: "numeric" })
    : "end of billing cycle";

  const handleSelectPlan = async () => {
    setIsLoading(true);
    try {
      if (plan.tier_key === "on_demand") {
        // On Demand: simple API call, no Stripe
        const res = await apiClient.post("/subscriptions/select-on-demand", {});
        if (!res.ok) {
          const data = await res.json();
          throw new Error(data?.detail?.detail || "Failed to switch plan");
        }
        queryClient.invalidateQueries({ queryKey: ["auth", "me"] });
        queryClient.invalidateQueries({ queryKey: ["subscription"] });
        toast.success("Plan updated to On Demand");
        router.push("/settings/billing");
      } else if (hasSubscription) {
        // Already subscribed: use plan change endpoint (dialog already confirmed)
        const res = await apiClient.post("/subscriptions/change", { plan_tier: plan.tier_key });
        if (!res.ok) {
          const data = await res.json();
          throw new Error(data?.detail?.detail || "Failed to change plan");
        }
        const result = await res.json();
        queryClient.invalidateQueries({ queryKey: ["auth", "me"] });
        queryClient.invalidateQueries({ queryKey: ["subscription"] });
        queryClient.invalidateQueries({ queryKey: ["credits"] });
        if (result.change_type === "upgrade") {
          toast.success(`Plan upgraded to ${result.new_plan}! New credits added.`);
        } else {
          toast.success(`Plan will change to ${result.new_plan} at end of billing cycle.`);
        }
        router.push("/settings/billing");
      } else {
        // New subscription: Stripe-hosted checkout redirect
        const res = await apiClient.post("/subscriptions/checkout", { plan_tier: plan.tier_key });
        if (!res.ok) {
          const data = await res.json();
          throw new Error(data?.detail?.detail || "Failed to create checkout");
        }
        const { checkout_url } = await res.json();
        window.location.href = checkout_url;
        return; // Don't reset loading since we're navigating away
      }
    } catch (err: any) {
      toast.error(err.message || "Something went wrong. Please try again.");
    } finally {
      setIsLoading(false);
    }
  };

  // Determine button label
  let buttonLabel: string;
  if (isCurrentPlan) {
    buttonLabel = "Current Plan";
  } else if (plan.tier_key === "on_demand") {
    buttonLabel = "Select On Demand";
  } else if (hasSubscription) {
    buttonLabel = isUpgrade
      ? `Upgrade to ${plan.display_name}`
      : `Downgrade to ${plan.display_name}`;
  } else {
    buttonLabel = `Subscribe to ${plan.display_name}`;
  }

  // For subscribed users changing plans, wrap button in confirmation dialog
  const needsConfirmation = hasSubscription && !isCurrentPlan && plan.tier_key !== "on_demand";

  const actionButton = (
    <Button
      className="w-full"
      variant={plan.is_popular && !isCurrentPlan ? "default" : "outline"}
      disabled={isCurrentPlan || isLoading}
      onClick={needsConfirmation ? undefined : handleSelectPlan}
    >
      {isLoading ? (
        <>
          <Loader2 className="mr-2 h-4 w-4 animate-spin" />
          Processing...
        </>
      ) : (
        buttonLabel
      )}
    </Button>
  );

  return (
    <Card className={plan.is_popular ? "border-l-[3px] border-l-primary relative" : ""}>
      <CardHeader className="space-y-1">
        <div className="flex items-center justify-between">
          <h2 className="text-xl font-semibold">{plan.display_name}</h2>
          <div className="flex gap-2">
            {plan.is_popular && (
              <Badge className="bg-primary text-primary-foreground">Most Popular</Badge>
            )}
            {isCurrentPlan && (
              <Badge variant="secondary">Current Plan</Badge>
            )}
          </div>
        </div>
        <p className="text-2xl font-semibold">
          {plan.price_display}
        </p>
      </CardHeader>
      <CardContent>
        <ul className="space-y-2 mb-6">
          {plan.features.map((feature) => (
            <li key={feature} className="flex items-start gap-2 text-sm text-muted-foreground">
              <Check className="h-4 w-4 mt-0.5 text-primary shrink-0" />
              <span>{feature}</span>
            </li>
          ))}
        </ul>

        {needsConfirmation ? (
          <AlertDialog>
            <AlertDialogTrigger asChild>
              {actionButton}
            </AlertDialogTrigger>
            <AlertDialogContent>
              <AlertDialogHeader>
                <AlertDialogTitle>
                  {isUpgrade ? `Upgrade to ${plan.display_name}?` : `Downgrade to ${plan.display_name}?`}
                </AlertDialogTitle>
                <AlertDialogDescription>
                  {isUpgrade
                    ? `You'll be charged the prorated difference immediately. Your new ${plan.credit_allocation} monthly credits will be added now.`
                    : `Your plan change takes effect at end of your billing cycle on ${periodEnd}. You'll continue on your current plan until then.`}
                </AlertDialogDescription>
              </AlertDialogHeader>
              <AlertDialogFooter>
                <AlertDialogCancel>Cancel</AlertDialogCancel>
                <AlertDialogAction onClick={handleSelectPlan} disabled={isLoading}>
                  {isLoading ? (
                    <><Loader2 className="mr-2 h-4 w-4 animate-spin" />Processing...</>
                  ) : (
                    isUpgrade ? "Confirm Upgrade" : "Confirm Downgrade"
                  )}
                </AlertDialogAction>
              </AlertDialogFooter>
            </AlertDialogContent>
          </AlertDialog>
        ) : (
          actionButton
        )}
      </CardContent>
    </Card>
  );
}
```
