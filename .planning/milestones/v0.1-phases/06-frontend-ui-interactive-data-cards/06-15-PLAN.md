**Step 3: Add useUpdateFileContext hook to useFileManager.ts (optional but clean)**

Add a reusable mutation hook after useDeleteFile:

```typescript
/**
 * TanStack Query mutation for updating file context (FILE-06)
 */
export function useUpdateFileContext() {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: async ({
      fileId,
      context,
    }: {
      fileId: string;
      context: string;
    }): Promise<FileDetailResponse> => {
      const response = await apiClient.post(`/files/${fileId}/context`, {
        context,
      });
      if (!response.ok) {
        throw new Error("Failed to update file context");
      }
      return response.json();
    },
    onSuccess: (_data, variables) => {
      // Invalidate summary to show updated user_context
      queryClient.invalidateQueries({
        queryKey: ["files", "summary", variables.fileId],
      });
    },
  });
}
```

Add the necessary import at top of useFileManager.ts: `import { FileDetailResponse } from "@/types/file";` (add FileDetailResponse to the existing import).

Then refactor FileInfoModal to use this hook instead of direct apiClient calls. This is cleaner and follows the existing pattern. Replace the inline handleSubmitFeedback with the mutation:
