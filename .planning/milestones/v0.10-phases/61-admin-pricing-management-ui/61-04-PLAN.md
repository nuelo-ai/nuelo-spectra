- Props:

  - `title`: varies by action type (see UI-SPEC copywriting contract)
  - `impactText`: varies by action type. Exact copy:
    - edit-subscription: `This will update the subscription price for {tier name}. A new Stripe Price will be created automatically. Existing subscribers will keep their current price.`
    - edit-package: `This will update the {package name} credit package. Changes take effect immediately for new purchases.`
    - reset-subscriptions: `This will reset all subscription prices to their config-file defaults. New Stripe Prices will be created if needed. Existing subscribers keep their current price.`
    - reset-packages: `This will reset all credit packages to config-file defaults. Names, prices, credits, and display order will be restored. Existing purchase history is preserved.`
  - `confirmLabel`: "Confirm Changes" for edits, "Confirm Reset" for resets
  - `variant`: "default" for edits, "destructive" for resets
  - `isPending`: from the relevant mutation's `.isPending`
  - `error`: `confirmError` state

```typescript
async function handleConfirm(password: string) {
  setConfirmError(null);
  try {
    if (confirmAction.type === "edit-subscription") {
      const { tier, priceCents } = confirmAction.payload;
      await updateBillingSettings.mutateAsync({
        [`price_${tier}_monthly_cents`]: priceCents,
        password,
      });
      toast.success("Pricing updated successfully");
      setEditingTier(null);
    } else if (confirmAction.type === "edit-package") {
      const { packageId, name, priceCents, creditAmount, displayOrder, isActive } = confirmAction.payload;
      await updateCreditPackage.mutateAsync({
        packageId,
        name,
        price_cents: priceCents,
        credit_amount: creditAmount,
        display_order: displayOrder,
        is_active: isActive,
        password,
      });
      toast.success("Pricing updated successfully");
      setEditingPackage(null);
    } else if (confirmAction.type === "reset-subscriptions") {
      await resetSubscriptionPricing.mutateAsync({ password });
      toast.success("Subscription pricing reset to defaults");
    } else if (confirmAction.type === "reset-packages") {
      await resetCreditPackages.mutateAsync({ password });
      toast.success("Credit packages reset to defaults");
    }
    setConfirmAction(null);
  } catch (e: any) {
    if (e.message === "Incorrect password") {
      setConfirmError("Incorrect password. Please try again.");
    } else {
      toast.error(e.message || "Failed to save changes. Please try again.");
      setConfirmAction(null);
    }
  }
}
```
