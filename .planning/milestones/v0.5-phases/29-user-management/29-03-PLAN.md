Add module-level in-memory challenge store:

```python
import secrets
import string
import time
import threading

# Challenge store: (admin_id_str, scope_key) -> (code, expires_at)
_challenge_store: dict[tuple[str, str], tuple[str, float]] = {}
_challenge_lock = threading.Lock()

def generate_challenge_code(admin_id: UUID, scope_key: str, ttl: int = 300) -> str:
    """Generate a 6-char alphanumeric challenge code. TTL defaults to 5 minutes."""
    chars = string.ascii_letters + string.digits
    code = "".join(secrets.choice(chars) for _ in range(6))
    key = (str(admin_id), scope_key)
    with _challenge_lock:
        _challenge_store[key] = (code, time.time() + ttl)
        # Cleanup expired entries
        expired = [k for k, (_, exp) in _challenge_store.items() if time.time() > exp]
        for k in expired:
            del _challenge_store[k]
    return code

def verify_challenge_code(admin_id: UUID, scope_key: str, submitted: str) -> bool:
    """Verify and consume a challenge code (single-use). Timing-safe comparison."""
    key = (str(admin_id), scope_key)
    with _challenge_lock:
        entry = _challenge_store.pop(key, None)
    if entry is None:
        return False
    code, expires_at = entry
    if time.time() > expires_at:
        return False
    return secrets.compare_digest(code, submitted)
```

async def delete_user(db: AsyncSession, user_id: UUID, admin_id: UUID) -> str:
    """Hard delete a user with audit anonymization and physical file cleanup."""
    from app.models.file import File
    from app.models.admin_audit_log import AdminAuditLog
    import os

    # Fetch user
    result = await db.execute(select(User).where(User.id == user_id))
    user = result.scalar_one_or_none()
    if not user:
        raise ValueError("User not found")

    # Cannot delete yourself
    if user_id == admin_id:
        raise ValueError("Cannot delete your own account")

    anon_label = f"deleted_user_{str(user_id)[:8]}"

    # Step 1: Anonymize existing audit log entries that reference this user
    # Update details JSON where target_id matches user_id
    audit_entries = await db.execute(
        select(AdminAuditLog).where(AdminAuditLog.target_id == str(user_id))
    )
    for entry in audit_entries.scalars().all():
        if entry.details:
            entry.details = {**entry.details, "_anonymized": anon_label}
        else:
            entry.details = {"_anonymized": anon_label}
        entry.target_id = anon_label

    # Step 2: Get physical file paths before deletion
    file_results = await db.execute(
        select(File.file_path).where(File.user_id == user_id)
    )
    file_paths = [row[0] for row in file_results.all()]

    # Step 3: Create deletion audit entry (BEFORE deleting)
    from app.services.admin.audit import log_admin_action
    await log_admin_action(
        db,
        admin_id=admin_id,
        action="delete_user",
        target_type="user",
        target_id=anon_label,
        details={
            "anonymized_as": anon_label,
            "email_domain": user.email.split("@")[1] if "@" in user.email else "unknown",
        },
    )

    # Step 4: Delete user (CASCADE handles related DB records)
    await db.delete(user)
    await db.flush()

    # Step 5: Delete physical files (best effort, after DB flush)
    for path in file_paths:
        try:
            if os.path.exists(path):
                os.remove(path)
        except OSError:
            pass  # File already gone or permission issue -- not critical

    return anon_label
```

async def bulk_activate(db: AsyncSession, user_ids: list[UUID]) -> dict:
    """Activate multiple users. Returns success/failure counts."""
    from app.dependencies import clear_user_deactivation
    succeeded, failed, errors = 0, 0, []
    for uid in user_ids:
        try:
            result = await db.execute(select(User).where(User.id == uid).with_for_update())
            user = result.scalar_one_or_none()
            if not user:
                errors.append({"user_id": str(uid), "error": "User not found"})
                failed += 1
                continue
            user.is_active = True
            clear_user_deactivation(uid)
            succeeded += 1
        except Exception as e:
            errors.append({"user_id": str(uid), "error": str(e)})
            failed += 1
    await db.flush()
    return {"succeeded": succeeded, "failed": failed, "errors": errors}

async def bulk_deactivate(db: AsyncSession, user_ids: list[UUID]) -> dict:
    """Deactivate multiple users with immediate token invalidation."""
    from app.dependencies import mark_user_deactivated
    succeeded, failed, errors = 0, 0, []
    for uid in user_ids:
        try:
            result = await db.execute(select(User).where(User.id == uid).with_for_update())
            user = result.scalar_one_or_none()
            if not user:
                errors.append({"user_id": str(uid), "error": "User not found"})
                failed += 1
                continue
            user.is_active = False
            mark_user_deactivated(uid)
            succeeded += 1
        except Exception as e:
            errors.append({"user_id": str(uid), "error": str(e)})
            failed += 1
    await db.flush()
    return {"succeeded": succeeded, "failed": failed, "errors": errors}

async def bulk_change_tier(db: AsyncSession, user_ids: list[UUID], new_class: str, admin_id: UUID) -> dict:
    """Change tier for multiple users (each gets credit reset)."""
    from app.services.admin.tiers import change_user_tier
    succeeded, failed, errors = 0, 0, []
    for uid in user_ids:
        try:
            await change_user_tier(db, uid, new_class, admin_id)
            succeeded += 1
        except ValueError as e:
            errors.append({"user_id": str(uid), "error": str(e)})
            failed += 1
    await db.flush()
    return {"succeeded": succeeded, "failed": failed, "errors": errors}

async def bulk_adjust_credits(
    db: AsyncSession, user_ids: list[UUID], amount: Decimal | None,
    delta: Decimal | None, reason: str, admin_id: UUID
) -> dict:
    """Bulk credit adjustment. Either set exact amount OR add/deduct delta."""
    from app.services.credit import CreditService
    from app.models.user_credit import UserCredit
    succeeded, failed, errors = 0, 0, []
    for uid in user_ids:
        try:
            if delta is not None:
                # Add/deduct delta from current balance
                await CreditService.admin_adjust(db, uid, delta, reason, admin_id)
            elif amount is not None:
                # Set exact amount: compute delta from current balance
                credit_result = await db.execute(
                    select(UserCredit).where(UserCredit.user_id == uid).with_for_update()
                )
                credit = credit_result.scalar_one_or_none()
                if credit is None:
                    errors.append({"user_id": str(uid), "error": "No credit record"})
                    failed += 1
                    continue
                actual_delta = amount - credit.balance
                await CreditService.admin_adjust(db, uid, actual_delta, reason, admin_id)
            succeeded += 1
        except Exception as e:
            errors.append({"user_id": str(uid), "error": str(e)})
            failed += 1
    await db.flush()
    return {"succeeded": succeeded, "failed": failed, "errors": errors}

async def bulk_delete(db: AsyncSession, user_ids: list[UUID], admin_id: UUID) -> dict:
    """Bulk hard delete users with anonymization."""
    succeeded, failed, errors = 0, 0, []
    for uid in user_ids:
        try:
            await delete_user(db, uid, admin_id)
            succeeded += 1
        except ValueError as e:
            errors.append({"user_id": str(uid), "error": str(e)})
            failed += 1
    return {"succeeded": succeeded, "failed": failed, "errors": errors}
```

    user_class: str

class BulkCreditAdjustRequest(BaseModel):
    user_ids: list[UUID] = Field(..., min_length=1, max_length=100)
    amount: Decimal | None = None   # Set exact amount (mutually exclusive with delta)
    delta: Decimal | None = None    # Add/deduct from current (mutually exclusive with amount)
    reason: str = Field(..., min_length=1, max_length=500)

    @model_validator(mode="after")
    def exactly_one_mode(self):
        if (self.amount is None) == (self.delta is None):
            raise ValueError("Provide exactly one of 'amount' (set exact) or 'delta' (add/deduct)")
        return self

class BulkDeleteRequest(BaseModel):
    user_ids: list[UUID] = Field(..., min_length=1, max_length=100)
    challenge_code: str = Field(..., min_length=6, max_length=6)

class BulkActionResult(BaseModel):
    succeeded: int
    failed: int
    errors: list[dict]  # [{user_id, error}]

**Individual Delete (USER-13):**

**Endpoint 1: `POST /{user_id}/delete-challenge` — Generate challenge code**
- Call `generate_challenge_code(current_admin.id, f"delete_user_{user_id}")`.
- Return `DeleteChallengeResponse(challenge_code=code, expires_in=300)`.

**Endpoint 2: `DELETE /{user_id}` — Confirm delete with challenge**
- Accept `body: DeleteConfirmRequest` with challenge_code.
- Verify: `verify_challenge_code(current_admin.id, f"delete_user_{user_id}", body.challenge_code)`. If fails, raise `HTTPException(400, "Invalid or expired challenge code")`.
- Call `delete_user(db, user_id, current_admin.id)`.
- Commit. (Audit log is created inside delete_user before the delete.)
- Return `{"message": "User deleted", "anonymized_as": anon_label}`.

**Bulk Operations:**

**Endpoint 3: `POST /bulk/activate`**
- Accept `body: BulkUserActionRequest`.
- Call `bulk_activate(db, body.user_ids)`.
- Audit log: `log_admin_action(db, admin_id=..., action="bulk_activate", target_type="users", details={"user_ids": [str(uid) for uid in body.user_ids], "result": result})`.
- Commit.
- Return `BulkActionResult`.

**Endpoint 4: `POST /bulk/deactivate`**
- Same pattern as activate, action="bulk_deactivate".
- Call `bulk_deactivate(db, body.user_ids)`.

**Endpoint 5: `POST /bulk/tier-change`**
- Accept `body: BulkTierChangeRequest`.
- Call `bulk_change_tier(db, body.user_ids, body.user_class, current_admin.id)`.
- Audit log with action="bulk_tier_change".
- Commit.

**Endpoint 6: `POST /bulk/credit-adjust`**
- Accept `body: BulkCreditAdjustRequest`.
- Call `bulk_adjust_credits(db, body.user_ids, body.amount, body.delta, body.reason, current_admin.id)`.
- Audit log with action="bulk_credit_adjust".
- Commit.

**Endpoint 7: `POST /bulk/delete-challenge` — Generate bulk delete challenge**
- Accept `body: BulkUserActionRequest` (just user_ids for display).
- Call `generate_challenge_code(current_admin.id, f"bulk_delete_{len(body.user_ids)}")`.
- Return `DeleteChallengeResponse`.

**Endpoint 8: `POST /bulk/delete` — Confirm bulk delete**
- Accept `body: BulkDeleteRequest` (user_ids + challenge_code).
- Verify challenge: `verify_challenge_code(current_admin.id, f"bulk_delete_{len(body.user_ids)}", body.challenge_code)`.
- Call `bulk_delete(db, body.user_ids, current_admin.id)`.
- Commit.
- Return `BulkActionResult`.

All endpoints inject `db: DbSession`, `current_admin: CurrentAdmin`, `request: Request`. Use `client_ip = request.client.host if request.client else None` for audit logging.
