From backend/app/services/collection.py (remove_file_from_collection pattern to follow):
```python
@staticmethod
async def remove_file_from_collection(
    db: AsyncSession,
    collection_id: UUID,
    file_id: UUID,
    user_id: UUID,
) -> bool:
    stmt = select(CollectionFile).where(
        CollectionFile.collection_id == collection_id,
        CollectionFile.file_id == file_id,
        CollectionFile.user_id == user_id,
    )
    result = await db.execute(stmt)
    cf = result.scalar_one_or_none()
    if cf is None:
        return False
    await db.delete(cf)
    await db.flush()
    return True
```
