        <h1 text-3xl font-semibold>Hi {name}! What would you like to analyze today?</h1>
        {!hasLinkedFiles && <p text-muted-foreground>Add a file to start exploring your data</p>}
        {hasLinkedFiles && suggestions && <QuerySuggestions ... />}
      </div>
      <div fixed-bottom chat-input-area>
        <ChatInput onSend={handleSend} disabled={false} placeholder={...} />
      </div>
